 14  enum terrain { //declares different parts of the grid
 15      empty,
 16      wall,
 17      goal,
 18      crumb

 21  void alloc_maze()
 22  {
 23      maze = malloc(rows * sizeof(char*));
 24  
 25      int i;
 26      for (i = 0; i < rows; ++i){
 27          maze[i] = malloc(cols * sizeof(char*));
 28      }
 29  }
 30  
 31  void alloc_visited()
 32  {
 33      visited = malloc(rows * sizeof(char*));
 34  
 35      int i;
 36      for (i = 0; i < rows; ++i){
 37          visited[i] = malloc(cols * sizeof(char*));
 38      }
 39  }

 41  void get_maze(char* file_name)
 42  {
 43      char c;
 44      char rows_s[3] = { '\0' };
 45      char cols_s[3] = { '\0' };
 46      int rows_i = 0;
 47      int cols_i = 0;
 48      int swap = 0;
 49  
 50      FILE* maze_file = fopen(file_name, "r");
 51  
 52      if (maze_file) {
 53          while ((c = getc(maze_file)) != EOF) {
 54              if (c == '\n') {
 55                  break;
 56              } else if (c == ','){
 57                  swap = 1;
 58              } else if (!swap) {
 59                  rows_s[rows_i] = c;
 60                  rows_i++;
 61              } else {
 62                  cols_s[cols_i] = c;
 63                  cols_i++;
 64              }
 65          }
 66      } else {
 67          printf("No such file!");
 68          return;
 69      }
 70  
 71      rows = atoi(rows_s);
 72      cols = atoi(cols_s);
 73  
 74      alloc_maze();
 75  
 76      int i,j;
 77  
 78      for (i = 0; i < rows; ++i) {
 79          for (j = 0; j < cols; ++j) {
 80  
 81              if ((c = getc(maze_file)) == '\n') {
 82                  c = getc(maze_file);
 83              }
 84  
 85              maze[i][j] = c;
 86  
 87              if (c =='S') {
 88                  start_row = i;
 89                  start_col = j;
 90              }
 91          }
 92      }
 93  
 94      fclose(maze_file);
 95  }

 97  void get_visited() {
 98      alloc_visited();
 99      for (int i = 0; i < rows; i++) {
100          for (int j = 0; j < cols; j++) {
101              if (maze[i][j] == '+') {
102                  visited[i][j] = wall;
103              } else if (maze[i][j] == 'G') {
104                  visited[i][j] = goal;
105              } else {
106                  visited[i][j] = empty;
107  
108              }
109          }
110      }
111  }

113  int dfs(int row, int col) {
114  
115      int* current = &visited[row][col];
116      /*printf("(%d, ", &current);
117      printf("%d)", &current);*/
118      testNum++;
119      if (*current == goal) {
120          return 1;
121      }
122  
123      if (*current == empty) {
124          *current = wall;
125  
126          if (dfs(row, col - 1)) {
127              *current = crumb; //looks one left and checks if it is an empty spot, if so place a crumb.
128              return 1;
129          }
130          if (dfs(row, col + 1)) {
131              *current = crumb; //looks one right and checks if it is an empty spot, if so place a crumb.
132              return 1;
133          }
134          if (dfs(row - 1, col)) {
135              *current = crumb; //looks one up and checks if it is an empty spot, if so place a crumb.
136              return 1;
137          }
138          if (dfs(row + 1, col)) {
139              *current = crumb; //looks one down and checks if it is an empty spot, if so place a crumb.
140  
141  
142              return 1;
143          }
144      }
145  
146      return 0;
147  }

150  void add_crumbs(){
151      for (int i = 0; i < rows; i++) {
152          for (int j = 0; j < cols; j++) {
153              //printf("(%d, %d)", i, j);
154              if (maze[i][j] != 'S') {
155                  if (visited[i][j] == crumb) {
156                      maze[i][j] = '@';
157  
158                  }
159              }
160          }
161      }
162  }
